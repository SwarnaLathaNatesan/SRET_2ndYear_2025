Matrix Programs in Python

Transpose

def transpose(matrix):
    result = []
    for i in range(len(matrix[0])):
        result.append([matrix[j][i] for j in range(len(matrix))])
    return result

Matrix Add

def matrix_add(A, B):
    result = []
    for i in range(len(A)):
        result.append([A[i][j] + B[i][j] for j in range(len(A[0]))])
    return result

Scalar Multiply

def scalar_multiply(matrix, scalar):
    result = []
    for row in matrix:
        result.append([element * scalar for element in row])
    return result

Trace

def trace(matrix):
    total = 0
    for i in range(min(len(matrix), len(matrix[0]))):
        total += matrix[i][i]
    return total

Is Square

def is_square(matrix):
    for row in matrix:
        if len(row) != len(matrix):
            return False
    return True

Main Diagonal

def main_diagonal(matrix):
    diagonal = []
    for i in range(min(len(matrix), len(matrix[0]))):
        diagonal.append(matrix[i][i])
    return diagonal

Anti Diagonal

def anti_diagonal(matrix):
    diagonal = []
    for i in range(len(matrix)):
        diagonal.append(matrix[i][len(matrix)-1-i])
    return diagonal

Row Sums

def row_sums(matrix):
    sums = []
    for row in matrix:
        sums.append(sum(row))
    return sums

Col Sums

def col_sums(matrix):
    sums = []
    for j in range(len(matrix[0])):
        sums.append(sum(matrix[i][j] for i in range(len(matrix))))
    return sums

Is Identity

def is_identity(matrix):
    for i in range(len(matrix)):
        for j in range(len(matrix[0])):
            if matrix[i][j] != (1 if i == j else 0):
                return False
    return True

Flatten Matrix

def flatten_matrix(matrix):
    flat = []
    for row in matrix:
        flat.extend(row)
    return flat

Identity Matrix

def identity_matrix(n):
    matrix = []
    for i in range(n):
        matrix.append([1 if i == j else 0 for j in range(n)])
    return matrix

Elementwise Multiply

def elementwise_multiply(A, B):
    result = []
    for i in range(len(A)):
        result.append([A[i][j] * B[i][j] for j in range(len(A[0]))])
    return result

Is Zero Matrix

def is_zero_matrix(matrix):
    for row in matrix:
        for element in row:
            if element != 0:
                return False
    return True

Matrix Dimensions

def matrix_dimensions(matrix):
    if not matrix:
        return (0, 0)
    return (len(matrix), len(matrix[0]))

Row Maximums

def row_maximums(matrix):
    maximums = []
    for row in matrix:
        maximums.append(max(row))
    return maximums

Col Maximums

def col_maximums(matrix):
    maximums = []
    for j in range(len(matrix[0])):
        maximums.append(max(matrix[i][j] for i in range(len(matrix))))
    return maximums

Rotate 180

def rotate_180(matrix):
    result = []
    for row in reversed(matrix):
        result.append(list(reversed(row)))
    return result

Is Symmetric

def is_symmetric(matrix):
    for i in range(len(matrix)):
        for j in range(i):
            if matrix[i][j] != matrix[j][i]:
                return False
    return True

Min Element Pos

def min_element_pos(matrix):
    min_val = min(min(row) for row in matrix)
    for i in range(len(matrix)):
        for j in range(len(matrix[0])):
            if matrix[i][j] == min_val:
                return (i, j)
